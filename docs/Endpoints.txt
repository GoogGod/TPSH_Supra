# Актуальные API эндпоинты (backend)
# Формат: METHOD<TAB>PATH<TAB>Описание<TAB>Доступ

POST	/api/v1/auth/login/	Получить access/refresh JWT	Публичный
POST	/api/v1/auth/token/refresh/	Обновить access по refresh	Публичный
POST	/api/v1/auth/register/	Регистрация пользователя	Auth (Manager, Admin)
POST	/api/v1/auth/logout/	Логаут (blacklist refresh)	Auth

GET	/api/v1/users/me/	Профиль текущего пользователя	Auth
GET	/api/v1/users/	Список пользователей (менеджер/сотрудник видит свой venue + admin)	Auth
GET	/api/v1/users/{id}/	Детали пользователя	Auth (Manager, Admin)
PATCH	/api/v1/users/{id}/	Редактирование пользователя	Auth (Manager, Admin)
PUT	/api/v1/users/{id}/	Полное обновление пользователя	Auth (Manager, Admin)

GET	/api/v1/venues/	Список активных объектов	Auth (Manager, Admin)
POST	/api/v1/venues/create/	Создать новый объект	Auth (Admin)

GET	/api/v1/schedule/status/	Статус расписания на период (с фильтром по venue)	Auth
POST	/api/v1/schedule/generate/	Сгенерировать черновик расписания из ML пайплайна	Auth (Manager, Admin)
POST	/api/v1/schedule/upload/	Загрузить CSV расписания вручную	Auth (Manager, Admin)

GET	/api/v1/schedule/monthly/	Список месячных расписаний	Auth
GET	/api/v1/schedule/monthly/{id}/	Детали расписания (слоты + записи)	Auth
PATCH	/api/v1/schedule/monthly/{id}/entries/bulk-update/	Массово править entries в draft	Auth (Manager, Admin)
POST	/api/v1/schedule/monthly/{id}/publish/	Опубликовать draft	Auth (Manager, Admin)
DELETE	/api/v1/schedule/monthly/{id}/delete/	Удалить draft	Auth (Manager, Admin)

POST	/api/v1/schedule/slots/{id}/claim/	Сотрудник занимает свободный слот	Auth (Employee)
POST	/api/v1/schedule/slots/{id}/assign/	Менеджер/админ назначает сотрудника в слот	Auth (Manager, Admin)
POST	/api/v1/schedule/slots/{id}/unassign/	Снять назначение со слота	Auth (Manager, Admin)

POST	/api/v1/forecast/upload-data/	Загрузить raw данные (csv/xlsx)	Auth (Admin)
POST	/api/v1/forecast/run/	Запустить ML pipeline	Auth (Manager*, Admin)
GET	/api/v1/forecast/runs/	Список запусков ML	Auth (Admin)
GET	/api/v1/forecast/runs/{id}/	Детали запуска ML	Auth (Admin)
GET	/api/v1/forecast/hourly/	Почасовой прогноз	Auth (Manager, Admin)
GET	/api/v1/forecast/daily/	Дневной агрегированный прогноз	Auth (Manager, Admin)
GET	/api/v1/forecast/accuracy/	Метрики точности модели	Auth (Admin)
POST	/api/v1/forecast/generate-schedule/	DEPRECATED (совместимость)	Auth (Admin)

GET	/api/v1/notifications/	Список уведомлений текущего пользователя	Auth
GET	/api/v1/notifications/unread-count/	Количество непрочитанных	Auth
POST	/api/v1/notifications/read-all/	Прочитать все уведомления	Auth
POST	/api/v1/notifications/{id}/read/	Прочитать одно уведомление	Auth
POST	/api/v1/notifications/{id}/confirm/	Подтвердить назначение	Auth (Employee)
POST	/api/v1/notifications/{id}/reject/	Отклонить назначение	Auth (Employee)

GET	/api/v1/schema/	OpenAPI schema	Публичный/служебный
GET	/api/v1/docs/	Swagger UI	Публичный/служебный

# Примечание к /forecast/run/
# Manager может запускать только make_forecast на уже обученной модели.
# process_data/train_model/evaluate доступны только Admin.
